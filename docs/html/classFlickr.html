Detailed Description

Main wrapper class.

Main wrapper class in flickr api, call all methods through this class
Example

 UiClass::UiClass()
 {
                ...
                flickr = new Flickr("myApikey","mysecretkey",this);
 
                connect(flickr->auth(),SIGNAL(frobFetched (const QString &, FlickrError*)),
                                this,SLOT(frobFetched (const QString &, FlickrError*)));

                flickr->auth()->getFrob();
 }

 void UiClass::frobFetched ( const QString &frob, FlickrError *err )
 {
                if(!err->code){
                        QUrl url = flickr->auth()->getAuthUrl(frob);
                        QDesktopServices::openUrl(url);
                }
 }
