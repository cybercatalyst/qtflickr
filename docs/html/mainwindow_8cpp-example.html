#include "mainwindow.h"

#include <flickr.h>
#include <QtGui>

MainWindow::MainWindow()
                : QMainWindow()
{
        // Initialising flickr class with valid apikey and apisecret
        flickr = new Flickr ( "...",
                              "...",
                              this );
        
        // connecting listner to the findByUsernameFetched() signal
        connect ( flickr->people(),
                  SIGNAL ( findByUsernameFetched ( FindUserData *,FlickrError * ) ),
                  this,
                  SLOT ( findByUsernameFetched ( FindUserData *,FlickrError * ) ) );

        this->constructUi();
}


MainWindow::~MainWindow()
{
}

void MainWindow::constructUi()
{
        // ui layout
        usernameEdit = new QLineEdit();
        respEdit = new QTextEdit();
        respEdit->setReadOnly ( true );

        QFormLayout *formLayout = new QFormLayout();
        formLayout->addRow ( "Find User by Username:", usernameEdit );
        formLayout->addRow ( "Response:", respEdit );

        QWidget *main = new QWidget();
        main->setLayout ( formLayout );

        this->setCentralWidget ( main );

        connect ( usernameEdit,SIGNAL ( editingFinished () ),this,SLOT ( findUser() ) );
}

void MainWindow::findUser()
{
        // Searching user
        flickr->people()->findByUsername ( usernameEdit->text() );
}

void MainWindow::findByUsernameFetched ( FindUserData *data, FlickrError *err )
{
        if ( !err->code )
        {
                // printing user data on the screen
                QString userData ( "User id: "+data->id+"\n" );
                userData.append ( "User nsid: "+data->nsid+"\n" );
                userData.append ( "Username: "+data->username+"\n" );
                respEdit->setPlainText ( userData );
        }
        else
        {
                respEdit->setPlainText ( err->msg );
        }
}
